# AttendeeAddJson

Model representing an attendee to be added to an event. The attendee's assigned status determines if this add action is a registration or an invite.

## Example Usage

```typescript
import { AttendeeAddJson } from "@cvent/sdk/models/components";

let value: AttendeeAddJson = {
  event: {
    id: "0a955598-376e-43d4-a796-91d6d5f41e3f",
  },
  contact: {
    id: "ea66a709-2a93-4fa0-8b83-cf4f62df9dc3",
  },
  invitationList: {
    id: "06da38f9-12da-4cb6-a581-7a5954f8e0e2",
  },
  referenceId: "refid-marketingwebsite",
  note: "Would like to sit near the front of the room for sessions.",
  administrator: {
    id: "7949c335-b5b2-46cf-8f5d-f6b21795df51",
  },
  admissionItem: {
    id: "7949c335-b5b2-46cf-8f5d-f6b21795df51",
  },
  visibility: "Private",
  answers: [
    {
      question: {
        id: "60f1d05b-e3ee-4d70-a7f3-0edb9a76af4a",
      },
      value: [
        "1234",
      ],
    },
  ],
  sendEmail: false,
  status: "Accepted",
  guest: true,
  invitedBy: "Planner-Invited",
  responseMethod: "Self-Responded",
};
```
