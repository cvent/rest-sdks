# ExistingVideoTextTrackInput

let value: ExistingVideoTextTrackInput = {

    id: "04ca6ae2-0dc3-487b-953e-86d6abbdf7d3",

  label: "English",
  language: "en-US",

  errorMessage: "Input file was empty",
