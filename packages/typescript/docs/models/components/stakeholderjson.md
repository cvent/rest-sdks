# StakeholderJson

Represents an event stakeholder.

## Example Usage

```typescript
import { StakeholderJson } from "@cvent/sdk/models/components";

let value: StakeholderJson = {
  firstName: "Henry",
  lastName: "Potterfield",
  company: "Cvent Inc",
  title: "Event stakeholder",
  email: "stakeholderemail@test.com",
  workPhone: "555-555-5555",
  homePhone: "555-555-5555",
  mobilePhone: "555-555-5555",
  deleted: false,
};
```
