# TravelBidAmenityJson

Amenity details for a travel bid.

## Example Usage

```typescript
import { TravelBidAmenityJson } from "@cvent/sdk/models/components";

let value: TravelBidAmenityJson = {
  type: "amenity",
  code: "parking_valet",
  category: "Parking",
  price: 100,
  discountAmount: 90,
  internalCost: 80,
  description: "Valet parking is available for all guests.",
  status: "requested",
};
```
