import { RFCDate } from "@cvent/sdk/types";

let value: TravelBidFairSeasonJson = {

  startDate: new RFCDate("2021-07-13"),
  endDate: new RFCDate("2021-07-13"),

  releasePeriod: 1,

    {
      propertyRoom: {
        id: "9ea95519-2fc5-49d8-991b-e954f49044fe",

      doubleRate: 101,

  ],
  status: "requested",
};
