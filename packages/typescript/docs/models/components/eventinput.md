# EventInput

Represents an event.

## Example Usage

```typescript
import { EventInput } from "@cvent/sdk/models/components";

let value: EventInput = {
  title: "Cvent Connect 2020",
  format: "Virtual",
  description: "Cvent annual user conference.",
  start: new Date("2020-02-03T13:00:00.000Z"),
  end: new Date("2020-02-07T17:00:00.000Z"),
  timezone: "America/New_York",
  venues: [
    {
      name: "Capital Convention Center",
      address: {
        address1: "Cvent Inc.",
        address2: "4001 West Parmer Lane",
        address3: "PO Box 123",
        city: "Austin",
        countryCode: "US",
        postalCode: "78727",
        regionCode: "TX",
      },
    },
  ],
  note: "This event will need some coordination with external systems.",
  languages: [
    "en-US",
  ],
  planners: [],
  type: "Conference",
};
```
