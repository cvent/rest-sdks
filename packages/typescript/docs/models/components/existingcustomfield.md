# ExistingCustomField

This is used to denote an existing custom field.

## Example Usage

```typescript
import { ExistingCustomField } from "@cvent/sdk/models/components";

let value: ExistingCustomField = {
  created: new Date("2017-01-02T02:00:00Z"),
  createdBy: "hporter",
  lastModified: new Date("2019-02-12T03:00:00Z"),
  lastModifiedBy: "hporter",
  category: "Contact",
  name: "What is a your favorite color?",
  code: "FAVORITE_COLOR_CODE",
  type: "Open Ended Text - One Line",
  details: {
    minimum: 2,
    maximum: 8,
  },
  order: 1,
  helpText: "Enter your favorite color.",
  defaultTagText: "Your Division",
};
```
