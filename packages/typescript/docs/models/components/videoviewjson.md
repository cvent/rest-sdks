# VideoViewJson

Details of a single video view

## Example Usage

```typescript
import { VideoViewJson } from "@cvent/sdk/models/components";

let value: VideoViewJson = {
  contact: {
    id: "123e4567-e89b-12d3-a456-426614174000",
  },
  solution: "EVENTS_PLUS",
  video: {
    id: "456e4567-e89b-12d3-a456-426614174000",
  },
  totalDuration: 233000,
  watchDuration: 153600,
  watchPercentage: 52,
};
```
