# SegmentAttendeeResponseJson

Represents the attendees that are associated/disassociated to/from a segment

```typescript
let value: SegmentAttendeeResponseJson = {
  created: new Date("2017-01-02T02:00:00Z"),
  createdBy: "hporter",
  lastModified: new Date("2019-02-12T03:00:00Z"),
  lastModifiedBy: "hporter",
  segment: {
    id: "b13497ea-c117-4d62-a17b-e6c14d7e49d4",
  },
  id: "103097a4-143d-11e5-9f99-d0a637ee0032",
  disassociated: false,
};
```
