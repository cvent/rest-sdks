# EventQuestionJson

Represents a question associated with an event.

## Example Usage

```typescript
import { EventQuestionJson } from "@cvent/sdk/models/components";

let value: EventQuestionJson = {
  created: new Date("2017-01-02T02:00:00Z"),
  createdBy: "hporter",
  lastModified: new Date("2019-02-12T03:00:00Z"),
  lastModifiedBy: "hporter",
  id: "04ca6ae2-0dc3-487b-953e-86d6abbdf7d3",
  event: {
    id: "04ca6ae2-0dc3-487b-953e-86d6abbdf7d3",
  },
  text: "Let us know why you can't attend.",
  code: "ONE_LINE_43534",
  tag: "All Admission Items",
  session: {
    id: "04ca6ae2-0dc3-487b-953e-86d6abbdf7d3",
  },
  admissionItem: {
    id: "06ca5ae2-0dc3-487b-953e-86d6abbdf7d3",
  },
  internalNote: "This question is for the event registration form.",
};
```
