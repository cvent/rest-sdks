# ContactHistoryJson

A single instance of change for a specific contact in the contact's change history.

## Example Usage

```typescript
import { ContactHistoryJson } from "@cvent/sdk/models/components";

let value: ContactHistoryJson = {
  updatedFieldName: "Last Name",
  oldValue: "Wilson",
  newValue: "James",
  updatedThroughName: "Address Book Profile",
  updatedDate: new Date("2023-08-03T14:51:00.777Z"),
  updatedBy: "John Doe",
};
```
