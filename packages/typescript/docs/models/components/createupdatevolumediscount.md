# CreateUpdateVolumeDiscount

Representation of a volume based discount to be created or updated.

## Example Usage

```typescript
import { CreateUpdateVolumeDiscount } from "@cvent/sdk/models/components";
import { RFCDate } from "@cvent/sdk/types";

let value: CreateUpdateVolumeDiscount = {
  name: "10% off",
  method: {
    type: "BY_PERCENTAGE",
    value: 10,
  },
  effectiveFrom: new RFCDate("2025-01-01"),
  effectiveTo: new RFCDate("2025-02-01"),
  note: "10% off for early birds.",
  thresholdType: "ALL",
  thresholdLimit: 10,
  includePrimaryRegistrant: true,
  type: "VOLUME_DISCOUNT",
};
```
