# ExistingCustomFieldInput

This is used to denote an existing custom field.

## Example Usage

```typescript
import { ExistingCustomFieldInput } from "@cvent/sdk/models/components";

let value: ExistingCustomFieldInput = {
  category: "Contact",
  name: "What is a your favorite color?",
  code: "FAVORITE_COLOR_CODE",
  type: "Open Ended Text - One Line",
  details: {
    choices: [
      {
        text: "What is your current designation?",
      },
    ],
  },
  helpText: "Enter your favorite color.",
  defaultTagText: "Your Division",
};
```
