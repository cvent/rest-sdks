# CurrencyConversionRateResponse

Currency conversion rate response for a currency in an account.

## Example Usage

```typescript
import { CurrencyConversionRateResponse } from "@cvent/sdk/models/components";
import { RFCDate } from "@cvent/sdk/types";

let value: CurrencyConversionRateResponse = {
  conversionRate: 5.2,
  startDate: new RFCDate("2020-02-07"),
  endDate: new RFCDate("2020-02-08"),
  id: "9463c74e-18c6-401a-a710-ae0f485bf959",
  baseCurrency: "USD",
  currency: "EUR",
  lastModified: new Date("2020-02-07T00:00:00.00Z"),
};
```
